612        private final byte[] basicHeader = { 'I', 'I', 42, 0, 8, 0, 0, 0, // File introducer and pointer to first IFD
613                0, 0 }; // Number of tags start with two

617        // Builds up the tiffheader based on the options passed through.
618        private void buildHeader(COSDictionary options)
619        {
620 
621            final int numOfTags = 10; // The maximum tags we'll fill
622            final int maxAdditionalData = 24; // The maximum amount of additional data
623                                              // outside the IFDs. (bytes)
624 
625            // The length of the header will be the length of the basic header (10)
626            // plus 12 bytes for each IFD, 4 bytes as a pointer to the next IFD (will be 0)
627            // plus the length of the additional data
628 
629            int ifdSize = 10 + (12 * numOfTags) + 4;
630            tiffheader = new byte[ifdSize + maxAdditionalData];
631            java.util.Arrays.fill(tiffheader, (byte) 0);
632            System.arraycopy(basicHeader, 0, tiffheader, 0, basicHeader.length);
633 
634            // Additional data outside the IFD starts after the IFD's and pointer to the next IFD (0)
635            additionalOffset = ifdSize;
636 
637            // Now work out the variable values from TIFF defaults,
638            // PDF Defaults and the Dictionary for this XObject
639            short cols = 1728;
640            short rows = 0;
641            short blackis1 = 0;
642            short comptype = 3; // T4 compression
643            long t4options = 0; // Will set if 1d or 2d T4
644 
645            COSArray decode = getDecode();
646            // we have to invert the b/w-values,
647            // if the Decode array exists and consists of (1,0)
648            if (decode != null && decode.getInt(0) == 1)
649            {
650                blackis1 = 1;
651            }
652            COSBase dicOrArrayParms = options.getDictionaryObject(COSName.DECODE_PARMS);
653            COSDictionary decodeParms = null;
654            if (dicOrArrayParms instanceof COSDictionary)
655            {
656                decodeParms = (COSDictionary) dicOrArrayParms;
657            }
658            else if (dicOrArrayParms instanceof COSArray)
659            {
660                COSArray parmsArray = (COSArray) dicOrArrayParms;
661                if (parmsArray.size() == 1)
662                {
663                    decodeParms = (COSDictionary) parmsArray.getObject(0);
664                }
665                else
666                {
667                    // else find the first dictionary with Row/Column info and use that.
668                    for (int i = 0; i < parmsArray.size() && decodeParms == null; i++)
669                    {
670                        COSDictionary dic = (COSDictionary) parmsArray.getObject(i);
671                        if (dic != null
672                                && (dic.getDictionaryObject(COSName.COLUMNS) != null || dic
673                                        .getDictionaryObject(COSName.ROWS) != null))
674                        {
675                            decodeParms = dic;
676                        }
677                    }
678                }
679            }
680 
681            if (decodeParms != null)
682            {
683                cols = (short) decodeParms.getInt(COSName.COLUMNS, cols);
684                rows = (short) decodeParms.getInt(COSName.ROWS, rows);
685                if (decodeParms.getBoolean(COSName.BLACK_IS_1, false))
686                {
687                    blackis1 = 1;
688                }
689                int k = decodeParms.getInt(COSName.K, 0); // Mandatory parm
690                if (k < 0)
691                {
692                    // T6
693                    comptype = 4;
694                }
695                if (k > 0)
696                {
697                    // T4 2D
698                    comptype = 3;
699                    t4options = 1;
700                }
701                // else k = 0, leave as default T4 1D compression
702            }
703 
704            // If we couldn't get the number of rows, use the main item from XObject
705            if (rows == 0)
706            {
707                rows = (short) options.getInt(COSName.HEIGHT, rows);
708            }
709 
710            // Now put the tags into the tiffheader
711            // These musn't exceed the maximum set above, and by TIFF spec should be sorted into
712            // Numeric sequence.
713 
714            addTag(256, cols); // Columns
715            addTag(257, rows); // Rows
716            addTag(259, comptype); // T6
717            addTag(262, blackis1); // Photometric Interpretation
718            addTag(273, tiffheader.length); // Offset to start of image data - updated below
719            addTag(279, options.getInt(COSName.LENGTH)); // Length of image data
720            addTag(282, 300, 1); // X Resolution 300 (default unit Inches) This is arbitary
721            addTag(283, 300, 1); // Y Resolution 300 (default unit Inches) This is arbitary
722            if (comptype == 3)
723            {
724                addTag(292, t4options);
725            }
726            addTag(305, "PDFBOX"); // Software generating image
727        }
