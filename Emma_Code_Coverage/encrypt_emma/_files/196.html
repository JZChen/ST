691        fileLen = pdfFile.length();

702            int off = 0;
703            int readBytes;
704            while (off < trailByteCount)
705            {
706                readBytes = fIn.read(buf, off, trailByteCount - off);

710                {

716        }

718        {
719            if (fIn != null)
720            {
721                try
722                {
723                    fIn.close();
724                }

770    /**
771     * Searches last appearance of pattern within buffer. Lookup before _lastOff
772     * and goes back until 0.
773     * 
774     * @param pattern pattern to search for
775     * @param buf buffer to search pattern in
776     * @param endOff offset (exclusive) where lookup starts at
777     * 
778     * @return start offset of pattern within buffer or <code>-1</code> if
779     *         pattern could not be found
780     */
781    protected int lastIndexOf(final char[] pattern, final byte[] buf, final int endOff)
782    {
783        final int lastPatternChOff = pattern.length - 1;
784 
785        int bufOff = endOff;
786        int patOff = lastPatternChOff;
787        char lookupCh = pattern[patOff];
788 
789        while (--bufOff >= 0)
790        {
791            if (buf[bufOff] == lookupCh)
792            {
793                if (--patOff < 0)
794                {
795                    // whole pattern matched
796                    return bufOff;
797                }
798                // matched current char, advance to preceding one
799                lookupCh = pattern[patOff];
800            }
801            else if (patOff < lastPatternChOff)
802            {
803                // no char match but already matched some chars; reset
804                lookupCh = pattern[patOff = lastPatternChOff];
805            }
806        }
807 
808        return -1;
809    }
