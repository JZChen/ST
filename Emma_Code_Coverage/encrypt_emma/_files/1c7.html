419        }
420        else if (defaultOverlayPage != null)
421        {
422            layoutPage = defaultOverlayPage;
423        }
424        else if (useAllOverlayPages)
425        {
426            int usePageNum = (pageNumber - 1) % numberOfOverlayPages;
427            layoutPage = specificPageOverlayPage.get(usePageNum);
428        }
429        if (layoutPage != null)
430        {
431            PDResources resources = page.findResources();
432            if (resources == null)
433            {
434                resources = new PDResources();
435                page.setResources(resources);
436            }
437            String xObjectId = createOverlayXObject(page, layoutPage, layoutPage.overlayContentStream);
438            array.add(createOverlayStream(page, layoutPage, xObjectId));
439        }
440    }

442    private String createOverlayXObject(PDPage page, LayoutPage layoutPage, COSStream contentStream)
443    {
444        PDXObjectForm xobjForm = new PDXObjectForm(contentStream);
445        xobjForm.setResources(new PDResources(layoutPage.overlayResources));
446        xobjForm.setFormType(1);
447        xobjForm.setBBox(layoutPage.overlayMediaBox.createRetranslatedRectangle());
448        xobjForm.setMatrix(new AffineTransform());
449        PDResources resources = page.findResources();
450        return resources.addXObject(xobjForm, XOBJECT_PREFIX);
451    }

453    private COSStream createOverlayStream(PDPage page, LayoutPage layoutPage, String xObjectId) throws IOException
454    {
455        // create a new content stream that executes the XObject content
456        PDRectangle pageMediaBox = page.getMediaBox();
457        float hShift = (pageMediaBox.getWidth() - layoutPage.overlayMediaBox.getWidth()) / 2.0f;
458        float vShift = (pageMediaBox.getHeight() - layoutPage.overlayMediaBox.getHeight()) / 2.0f;
459        StringBuilder overlayStream = new StringBuilder();
460        overlayStream.append("q\nq 1 0 0 1 ");
461        overlayStream.append(float2String(hShift));
462        overlayStream.append(" ");
463        overlayStream.append(float2String(vShift) );
464        overlayStream.append(" cm /");
465        overlayStream.append(xObjectId);
466        overlayStream.append(" Do Q\nQ\n");
467        return createStream(overlayStream.toString());
468    }

470    private String float2String(float floatValue)
471    {
472        // use a BigDecimal as intermediate state to avoid 
473        // a floating point string representation of the float value
474        BigDecimal value = new BigDecimal(String.valueOf(floatValue));
475        String stringValue = value.toPlainString();
476        // remove fraction digit "0" only
477        if (stringValue.indexOf('.') > -1 && !stringValue.endsWith(".0"))
478        {
479            while (stringValue.endsWith("0") && !stringValue.endsWith(".0"))
480            {
481                stringValue = stringValue.substring(0,stringValue.length()-1);
482            }
483        }
484        return stringValue;
485    }

487    private COSStream createStream(String content) throws IOException
488    {
489        COSStream stream = new COSStream(new RandomAccessBuffer());
490        OutputStream out = stream.createUnfilteredStream();
491        out.write(content.getBytes("ISO-8859-1"));
492        out.close();
493        stream.setFilters(COSName.FLATE_DECODE);
494        return stream;
495    }
