812     * <p>
813     * example: 0 1 2 5 6 7 8 10
814     * <p>
815     * will create a array with follow ranges
816     * <p>
817     * 0 3 5 4 10 1
818     * <p>
819     * this mean that the element 0 is followed by two other related numbers 
820     * that represent a cluster of the size 3. 5 is follow by three other
821     * related numbers and create a cluster of size 4. etc.
822     * 
823     * @param xRefEntriesList list with the xRef entries that was written
824     * @return a integer array with the ranges

827    {

831 
832        ArrayList<Integer> list = new ArrayList<Integer>();
833        for( Object object : xRefEntriesList )
834        {
835            nr = (int)((COSWriterXRefEntry)object).getKey().getNumber();

837            {

840            }

842            {
843                last = nr;
844            }
845            else

848                list.add(count);
