492    {
493        BufferedImage alphaImage = new BufferedImage(bi.getWidth(), bi.getHeight(), alphaImageType);
494        for (int x = 0, w = bi.getWidth(); x < w; ++x)
495        {
496            for (int y = 0, h = bi.getHeight(); y < h; ++y)
497            {
498                int rgb = bi.getRGB(x, y);
499                int alpha = (rgb >>> 24);
500                alphaImage.setRGB(x, y, alpha | alpha << 8 | alpha << 16);
501            }
502        }
503        return alphaImage;
504    }

507    BufferedImage extractAlphaImage(BufferedImage bi)
508    {
509        int imageType = bi.getTransparency() == Transparency.BITMASK ? 
510                BufferedImage.TYPE_BYTE_BINARY : 
511                BufferedImage.TYPE_BYTE_GRAY;
512        WritableRaster alphaRaster = bi.getAlphaRaster();
513        if (alphaRaster == null)
514        {
515            // happens sometimes (PDFBOX-2654) despite colormodel claiming to have alpha
516            return createAlphaFromImage(bi, imageType);
517        }
518        BufferedImage alphaImage = new BufferedImage(alphaRaster.getWidth(), 
519                alphaRaster.getHeight(), 
520                imageType);
521        alphaImage.setData(alphaRaster);
522        return alphaImage;
