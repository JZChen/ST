409     * @throws IOException
410     */
411    protected COSBase readObject() throws IOException {
412        consumeWhitespace();
413        String string = readWord();
414        if(string.startsWith("<<")) {
415            // this is a dictionary
416            COSDictionary dictionary = new COSDictionary();
417            boolean atEndOfDictionary = false;
418            // remove the marker for the beginning of the dictionary
419            string = string.replaceAll("^<<", "");
420            
421            if("".equals(string) || string.matches("^\\w$"))
422                string = readWord().trim();
423            while(!atEndOfDictionary) {
424                COSName name = COSName.getPDFName(string);
425                COSBase object = readObject();
426                dictionary.setItem(name, object);
427 
428                byte singleByte = consumeWhitespace();
429                if(singleByte == '>') {
430                    readByte(); // get rid of the second '>'
431                    atEndOfDictionary = true;
432                }
433                if(!atEndOfDictionary)
434                    string = readWord().trim();
435            }
436            return dictionary;
437        } else if(string.startsWith("/")) {
438            // it's a dictionary label. i.e. /Type or /Pages or something similar
439            COSBase name = COSName.getPDFName(string);
440            return name;
441        } else if(string.startsWith("-")) {
442            // it's a negitive number
443            return parseNumber(string);
444        } else if(string.charAt(0) >= '0' && string.charAt(0) <= '9' ) {
445            // it's a COSInt or COSFloat, or a weak reference (i.e. "3 0 R")
446            // we'll have to peek ahead a little to see if it's a reference or not
447            long tempOffset = this.currentOffset;
448            consumeWhitespace();
449            String tempString = readWord();
450            if(tempString.matches("^[0-9]+$")) {
451                // it is an int, might be a weak reference...
452                tempString = readWord();
453                if(!"R".equals(tempString)) {
454                    // it's just a number, not a weak reference
455                    this.currentOffset = tempOffset;
456                    return parseNumber(string);
457                }
458            } else {
459                // it's just a number, not a weak reference
460                this.currentOffset = tempOffset;
461                return parseNumber(string);
462            }
463 
464            // it wasn't a number, so we need to parse the weak-reference
465            this.currentOffset = tempOffset;
466            int number = Integer.parseInt(string);
467            int gen = readInt();
468            String r = readWord();
469 
470            if(!"R".equals(r))
471                if(throwNonConformingException)
472                    throw new AssertionError("Expected keyword 'R' but found " + r);
473 
474            if(recursivlyRead) {
475                // seek to the object, read it, seek back to current location
476                long tempLocation = this.currentOffset;
477                this.currentOffset = this.xrefEntries.get(number).getByteOffset();
478                COSBase returnValue = readObject(number, gen);
479                this.currentOffset = tempLocation;
480                return returnValue;
481            } else {
482                // Put a COSUnknown there as a placeholder
483                COSObject obj = new COSObject(new COSUnread());
484                obj.setObjectNumber(COSInteger.get(number));
485                obj.setGenerationNumber(COSInteger.get(gen));
486                return obj;
487            }
488        } else if(string.startsWith("]")) {
489            // end of an array, just return null
490            if("]".equals(string))
491                return null;
492            int oldLength = string.length();
493            this.currentOffset -= oldLength;
494            return null;
495        } else if(string.startsWith("[")) {
496            // array of values
497            // we'll just pay attention to the first part (this is in case there
498            // is no whitespace between the "[" and the first element)
499            int oldLength = string.length();
500            string = "[";
501            this.currentOffset -= (oldLength - string.length() + 1);
502 
503            COSArray array = new COSArray();
504            COSBase object = readObject();
505            while(object != null) {
506                array.add(object);
507                object = readObject();
508            }
509            return array;
510        } else if(string.startsWith("(")) {
511            // this is a string (not hex encoded), strip off the '(' and read until ')'
512            StringBuilder sb = new StringBuilder(string.substring(1));
513            byte singleByte = readByte();
514            while(singleByte != ')') {
515                sb.append((char)singleByte);
516                singleByte = readByte();
517            }
518            return new COSString(sb.toString());
519        } else {
520            throw new RuntimeException("Not yet implemented: " + string
521                    + " loation=" + this.currentOffset);
522        }
523    }
524 
525    /**
526     * This will read the next string from the stream.
527     * @return The string that was read from the stream.
528     * @throws IOException If there is an error reading from the stream.
529     */
530    @Override
531    protected String readString() throws IOException {
532        consumeWhitespace();
533        StringBuilder buffer = new StringBuilder();
534        int c = pdfSource.read();
535        while(!isEndOfName((char)c) && !isClosing(c) && c != -1) {
536            buffer.append( (char)c );
537            c = pdfSource.read();
538        }
539        if (c != -1) {
540            pdfSource.unread(c);
541        }
542        return buffer.toString();
543    }

545    protected COSDictionary readDictionaryBackwards() throws IOException {
546        COSDictionary dict = new COSDictionary();
547        
548        // consume the last two '>' chars which signify the end of the dictionary
549        consumeWhitespaceBackwards();
550        byte singleByte = readByteBackwards();
551        if(throwNonConformingException) {
552            if(singleByte != '>')
553                throw new AssertionError("");
554        }
555        singleByte = readByteBackwards();
556        if(throwNonConformingException) {
557            if(singleByte != '>')
558                throw new AssertionError("");
559        }
560        
561        // check to see if we're at the end of the dictionary
562        boolean atEndOfDictionary = false;
563        singleByte = consumeWhitespaceBackwards();
564        if(singleByte == '<') {
565            inputFile.seek(currentOffset-1);
566            atEndOfDictionary =  ((byte)inputFile.read()) == '<';
567        }
568 
569        COSDictionary backwardsDictionary = new COSDictionary();
570        // while we're not at the end of the dictionary, read in entries
571        while(!atEndOfDictionary) {
572            COSBase object = readObjectBackwards();
573            COSName name = readNameBackwards();
574            backwardsDictionary.setItem(name, object);
575            
576            singleByte = consumeWhitespaceBackwards();
577            if(singleByte == '<') {
578                inputFile.seek(currentOffset-1);
579                atEndOfDictionary =  ((byte)inputFile.read()) == '<';
580            }
581        }
582 
583        // the dictionaries preserve the order keys were added, as such we shall
584        // add them in the proper order, not the reverse order
585        Set<COSName> backwardsKeys = backwardsDictionary.keySet();
586        for(int i = backwardsKeys.size()-1; i >=0; i--)
587            dict.setItem((COSName)backwardsKeys.toArray()[i], backwardsDictionary.getItem((COSName)backwardsKeys.toArray()[i]));
588        
589        // consume the last two '<' chars
590        readByteBackwards();
591        readByteBackwards();
592 
593        return dict;
594    }

598     * backwards until it finds a newline.  This should only be used if we are
599     * certain that the data will only be text, and not binary data.

602     * @throws IOException if there was an error reading data from the file
603     */

658    protected String readWord() throws IOException {
659        StringBuilder sb = new StringBuilder();
660        boolean stop = true;
661        do {
662            byte singleByte = readByte();
663            stop = this.isWhitespace(singleByte);
664 
665            // there are some additional characters which indicate the next element/word has begun
666            // ignore the first char we read, b/c the first char is the beginnging of this object, not the next one
667            if(!stop && sb.length() > 0) {
668                stop = singleByte == '/' || singleByte == '['
669                        || singleByte == ']'
670                        || (singleByte == '>' && !">".equals(sb.toString()));
671                if(stop) // we're stopping on a non-whitespace char, decrement the
672                    this.currentOffset--; // counter so we don't miss this character
673            }
674            if(!stop)
675                sb.append((char)singleByte);
676        } while(!stop);
677 
678        return sb.toString();
679    }
