622    /**

624     * we find the keyword "endstream" meaning we're at the end of this
625     * object. Some pdf files, however, forget to write some endstream tags
626     * and just close off objects with an "endobj" tag so we have to handle
627     * this case as well.
628     *
629     * This method is optimized using buffered IO and reduced number of
630     * byte compare operations.
631     *
632     * @param out  stream we write out to.
633     *
634     * @throws IOException
635     */
636    protected void readUntilEndStream( final OutputStream out ) throws IOException
637    {
638 
639        int bufSize;
640        int charMatchCount = 0;
641        byte[] keyw = ENDSTREAM;
642 
643        final int quickTestOffset = 5;  // last character position of shortest keyword ('endobj')
644 
645        // read next chunk into buffer; already matched chars are added to beginning of buffer
646        while ( ( bufSize = pdfSource.read( strmBuf, charMatchCount, strmBufLen - charMatchCount ) ) > 0 )
647        {
648            bufSize += charMatchCount;
649 
650            int bIdx = charMatchCount;
651            int quickTestIdx;
652 
653            // iterate over buffer, trying to find keyword match
654            for ( int maxQuicktestIdx = bufSize - quickTestOffset; bIdx < bufSize; bIdx++ )
655            {
656                // reduce compare operations by first test last character we would have to
657                // match if current one matches; if it is not a character from keywords
658                // we can move behind the test character;
659                // this shortcut is inspired by the Boyer-Moore string search algorithm
660                // and can reduce parsing time by approx. 20%
661                if ( ( charMatchCount == 0 ) &&
662                         ( ( quickTestIdx = bIdx + quickTestOffset ) < maxQuicktestIdx ) )
663                {
664 
665                    final byte ch = strmBuf[quickTestIdx];
666                    if ( ( ch > 't' ) || ( ch < 'a' ) )
667                    {
668                        // last character we would have to match if current character would match
669                        // is not a character from keywords -> jump behind and start over
670                        bIdx = quickTestIdx;
671                        continue;
672                    }
673                }
674 
675                final byte ch = strmBuf[bIdx];  // could be negative - but we only compare to ASCII
676 
677                if ( ch == keyw[ charMatchCount ] )
678                {
679                    if ( ++charMatchCount == keyw.length )
680                    {
681                        // match found
682                        bIdx++;
683                        break;
684                    }
685                }
686                else
687                {
688                    if ( ( charMatchCount == 3 ) && ( ch == ENDOBJ[ charMatchCount ] ) )
689                    {
690                        // maybe ENDSTREAM is missing but we could have ENDOBJ
691                        keyw = ENDOBJ;
692                        charMatchCount++;
693                    }
694                    else
695                    {
696                        // no match; incrementing match start by 1 would be dumb since we already know matched chars
697                        // depending on current char read we may already have beginning of a new match:
698                        // 'e': first char matched;
699                        // 'n': if we are at match position idx 7 we already read 'e' thus 2 chars matched
700                        // for each other char we have to start matching first keyword char beginning with next
701                        // read position
702                        charMatchCount = ( ch == E ) ? 1 : ( ( ch == N ) && ( charMatchCount == 7 ) ) ? 2 : 0;

704                        keyw = ENDSTREAM;
705                    }
706                }
707            }  // for
708 
709            int contentBytes = Math.max( 0, bIdx - charMatchCount );
710 
711            // write buffer content until first matched char to output stream
712            if ( contentBytes > 0 )
713            {
714                out.write( strmBuf, 0, contentBytes );
715            }
716            if ( charMatchCount == keyw.length )
717            {
718                // keyword matched; unread matched keyword (endstream/endobj) and following buffered content
719                pdfSource.unread( strmBuf, contentBytes, bufSize - contentBytes );
720                break;
721            }
722            else
723            {
724                // copy matched chars at start of buffer
725                System.arraycopy( keyw, 0, strmBuf, 0, charMatchCount );
726            }
727 
728        }  // while
729        
730        out.flush(); // this writes a lonely CR or drops trailing CR LF and LF
731    }
