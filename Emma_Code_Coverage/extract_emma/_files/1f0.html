436            // observed where a font has a space character with a character code
437            // other than 32, and where word spacing (Tw) was used.  In these cases,
438            // applying word spacing to either the non-32 space or to the character
439            // code 32 non-space resulted in errors consistent with this interpretation.
440            //
441            float spacingText = 0;
442            if( (string[i] == 0x20) && codeLength == 1 )
443            {
444                spacingText += wordSpacingText;
445            }
446            textMatrix.multiply(ctm, textXctm);
447            // Convert textMatrix to display units
448            // We need to instantiate a new Matrix instance here as it is passed to the TextPosition constructor below.
449            Matrix textMatrixStart = textStateParameters.multiply(textXctm);
450            
451            // TODO : tx should be set for horizontal text and ty for vertical text
452            // which seems to be specified in the font (not the direction in the matrix).
453            float tx = ((characterHorizontalDisplacementText)*fontSizeText)*horizontalScalingText;
454            float ty = 0;
455            // reset the matrix instead of creating a new one
456            td.reset();
457            td.setValue( 2, 0, tx );
458            td.setValue( 2, 1, ty );
459 
460            // The text matrix gets updated after each glyph is placed.  The updated
461            // version will have the X and Y coordinates for the next glyph.
462            // textMatrixEnd contains the coordinates of the end of the last glyph without 
463            // taking characterSpacingText and spacintText into account, otherwise it'll be
464            // impossible to detect new words within text extraction
465            textStateParameters.multiply(td, tempMatrix);
466            tempMatrix.multiply(textXctm, textMatrixEnd);
467            final float endXPosition = textMatrixEnd.getXPosition();
468            final float endYPosition = textMatrixEnd.getYPosition();
469 
470            // add some spacing to the text matrix (see comment above)
471            tx = ((characterHorizontalDisplacementText)*fontSizeText+characterSpacingText+spacingText)
472                    *horizontalScalingText;
473            td.setValue( 2, 0, tx );
474            td.multiply(textMatrix, textMatrix );
475            
476            // determine the width of this character
477            // XXX: Note that if we handled vertical text, we should be using Y here
478            float startXPosition = textMatrixStart.getXPosition();
479            float widthText = endXPosition - startXPosition;
480 
481            //there are several cases where one character code will
482            //output multiple characters.  For example "fi" or a
483            //glyphname that has no mapping like "visiblespace"
484            if( c != null )
485            {
486                validCharCnt++;
487            }
488            else 
489            {
490                // PDFBOX-373: Replace a null entry with "?" so it is
491                // not printed as "(null)"
492                c = "?";
493            }
494            totalCharCnt++;
495 
496            float totalVerticalDisplacementDisp = maxVerticalDisplacementText * fontSizeText * textXctm.getYScale();
497 
498            // process the decoded text
499            processTextPosition(
500                    new TextPosition(
501                            pageRotation,
502                            pageWidth,
503                            pageHeight,
504                            textMatrixStart,
505                            endXPosition,
506                            endYPosition,
507                            totalVerticalDisplacementDisp,
508                            widthText,
509                            spaceWidthDisp,
510                            c,
511                            codePoints,
512                            font,
513                            fontSizeText,
514                            (int)(fontSizeText * textMatrix.getXScale())
515                            ));
516        }
517    }

541     * This is used to handle an operation.
542     *
543     * @param operator The operation to perform.
544     * @param arguments The list of arguments.
545     *
546     * @throws IOException If there is an error processing the operation.
547     */
548    protected void processOperator( PDFOperator operator, List<COSBase> arguments ) throws IOException
549    {
550        try
551        {
552            String operation = operator.getOperation();
553            OperatorProcessor processor = (OperatorProcessor)operators.get( operation );
554            if( processor != null )
555            {
556                processor.setContext(this);
557                processor.process( operator, arguments );
558            }
559            else
560            {
561                if (!unsupportedOperators.contains(operation)) 
562                {
563                    LOG.info("unsupported/disabled operation: " + operation);
564                    unsupportedOperators.add(operation);
565                }
566            }
567        }
568        catch (Exception e)
569        {
570            LOG.warn(e, e);
571        }
572    }
