450     * This will return a list of signature fields.
451     *
452     * @param onlyEmptyFields only empty signature fields will be returned
453     * @return list of signature dictionaries as COSDictionary
454     * @throws IOException if no document catalog can be found
455     */
456    public List<COSDictionary> getSignatureFields(boolean onlyEmptyFields) throws IOException
457    {
458        COSObject documentCatalog = getCatalog();
459        if (documentCatalog != null)
460        {
461            COSDictionary acroForm = (COSDictionary)documentCatalog.getDictionaryObject(COSName.ACRO_FORM);
462            if (acroForm != null)
463            {
464                COSArray fields = (COSArray)acroForm.getDictionaryObject(COSName.FIELDS);
465                if (fields != null)
466                {
467                    // Some fields may contain twice references to a single field. 
468                    // This will prevent such double entries.
469                    HashMap<COSObjectKey, COSDictionary> signatures = new HashMap<COSObjectKey, COSDictionary>();
470                    for ( Object object : fields )
471                    {
472                        COSObject dict = (COSObject)object;
473                        if (COSName.SIG.equals(dict.getItem(COSName.FT)))
474                        {
475                            COSBase dictionaryObject = dict.getDictionaryObject(COSName.V);
476                            if (dictionaryObject == null || (dictionaryObject != null && !onlyEmptyFields))
477                            {
478                                signatures.put(new COSObjectKey(dict), (COSDictionary)dict.getObject());
479                            }
480                        }
481                    }
482                    return new LinkedList<COSDictionary>(signatures.values());
483                }
484            }
485        }
486        return Collections.emptyList();
487    }

671     * This method will search the list of objects for types of ObjStm.  If it finds
672     * them then it will parse out all of the objects from the stream that is contains.
673     *
674     * @throws IOException If there is an error parsing the stream.
675     */
676    public void dereferenceObjectStreams() throws IOException
677    {
678        for( COSObject objStream : getObjectsByType( COSName.OBJ_STM ) )
679        {
680            COSStream stream = (COSStream)objStream.getObject();
681            PDFObjectStreamParser parser =
682                new PDFObjectStreamParser(stream, this, forceParsing);
683            parser.parse();
684            for( COSObject next : parser.getObjects() )
685            {
686                COSObjectKey key = new COSObjectKey( next );
687                if ( objectPool.get(key) == null || objectPool.get(key).getObject() == null ||
688                     // xrefTable stores negated objNr of objStream for objects in objStreams
689                     (xrefTable.containsKey(key) && xrefTable.get(key) == -objStream.getObjectNumber().longValue()) )
690                {
691                    COSObject obj = getObjectFromPool(key);
692                    obj.setObject(next.getObject());
693                }
694            }
695        }
696    }
