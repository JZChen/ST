412    private float[] calculateInputValues(int x, int y)
413    {
414        // According to Adobes Technical Note #5600 we have to do the following
415        //
416        // x0, y0, r0 defines the start circle x1, y1, r1 defines the end circle
417        //
418        // The parametric equations for the center and radius of the gradient fill circle moving
419        // between the start circle and the end circle as a function of s are as follows:
420        //
421        // xc(s) = x0 + s * (x1 - x0) yc(s) = y0 + s * (y1 - y0) r(s) = r0 + s * (r1 - r0)
422        //
423        // Given a geometric coordinate position (x, y) in or along the gradient fill, the
424        // corresponding value of s can be determined by solving the quadratic constraint equation:
425        //
426        // [x - xc(s)]2 + [y - yc(s)]2 = [r(s)]2
427        //
428        // The following code calculates the 2 possible values of s
429        //
430        double p = -(x - coords[0]) * x1x0 - (y - coords[1]) * y1y0 - coords[2] * r1r0;
431        double q = (Math.pow(x - coords[0], 2) + Math.pow(y - coords[1], 2) - r0pow2);
432        double root = Math.sqrt(p * p - denom * q);
433        float root1 = (float) ((-p + root) / denom);
434        float root2 = (float) ((-p - root) / denom);
435        if (denom < 0)
436        {
437            return new float[]
438            {
439                root1, root2
440            };
441        }
442        else
443        {
444            return new float[]
445            {
446                root2, root1
447            };
448        }
