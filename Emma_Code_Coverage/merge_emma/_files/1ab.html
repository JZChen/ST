498            pdfSource.unread(binCharTestArr, 0, readBytes);
499        }
500        if (!noBinData)
501        {
502            LOG.warn("ignoring 'EI' assumed to be in the middle of inline image");
503        }
504        return noBinData;
505    }

507    /**
508     * This will read an operator from the stream.
509     *
510     * @return The operator that was read from the stream.
511     *
512     * @throws IOException If there is an error reading from the stream.
513     */
514    protected String readOperator() throws IOException
515    {
516        skipSpaces();
517 
518        //average string size is around 2 and the normal string buffer size is
519        //about 16 so lets save some space.
520        StringBuffer buffer = new StringBuffer(4);
521        int nextChar = pdfSource.peek();
522        while(
523            nextChar != -1 && // EOF
524            !isWhitespace(nextChar) &&
525            !isClosing(nextChar) &&
526            nextChar != '[' &&
527            nextChar != '<' &&
528            nextChar != '(' &&
529            nextChar != '/' &&
530            (nextChar < '0' ||
531             nextChar > '9' ) )
532        {
533            char currentChar = (char)pdfSource.read();
534            nextChar = pdfSource.peek();
535            buffer.append( currentChar );
536            // Type3 Glyph description has operators with a number in the name
537            if (currentChar == 'd' && (nextChar == '0' || nextChar == '1') ) 
538            {
539                buffer.append( (char)pdfSource.read() );
540                nextChar = pdfSource.peek();
541            }
542        }
543        return buffer.toString();
544    }
