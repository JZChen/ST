475     *
476     * @param password The password to compute the encrypted key.
477     * @param o The o entry of the encryption dictionary.
478     * @param permissions The permissions for the document.
479     * @param id The document id.

481     * @param length The length of the encryption key.
482     * @param encryptMetadata The encryption metadata
483     *
484     * @return The encrypted key bytes.
485     *
486     * @throws CryptographyException If there is an error with encryption.
487     */
488    public final byte[] computeEncryptedKey(
489            byte[] password,
490            byte[] o,
491            int permissions,
492            byte[] id,
493            int encRevision,
494            int length,
495            boolean encryptMetadata)
496            throws CryptographyException
497        {
498            byte[] result = new byte[ length ];
499            try
500            {
501                //PDFReference 1.4 pg 78
502                //step1
503                byte[] padded = truncateOrPad( password );
504 
505                //step 2
506                MessageDigest md = MessageDigest.getInstance("MD5");
507                md.update( padded );
508 
509                //step 3
510                md.update( o );
511 
512                //step 4
513                byte zero = (byte)(permissions >>> 0);
514                byte one = (byte)(permissions >>> 8);
515                byte two = (byte)(permissions >>> 16);
516                byte three = (byte)(permissions >>> 24);
517 
518                md.update( zero );
519                md.update( one );
520                md.update( two );
521                md.update( three );
522 
523                //step 5
524                md.update( id );
525                
526                //(Security handlers of revision 4 or greater) If document metadata is not being encrypted, 
527                //pass 4 bytes with the value 0xFFFFFFFF to the MD5 hash function.
528                //see 7.6.3.3 Algorithm 2 Step f of PDF 32000-1:2008
529                if( encRevision == 4 && !encryptMetadata)
530                {
531                    md.update(new byte[]{(byte)0xff, (byte)0xff, (byte)0xff, (byte)0xff});
532                }
533                
534                byte[] digest = md.digest();
535 
536                //step 6
537                if( encRevision == 3 || encRevision == 4)
538                {
539                    for( int i=0; i<50; i++ )
540                    {
541                        md.reset();
542                        md.update( digest, 0, length );
543                        digest = md.digest();
544                    }
545                }
546 
547                //step 7
548                if( encRevision == 2 && length != 5 )
549                {
550                    throw new CryptographyException(
551                        "Error: length should be 5 when revision is two actual=" + length );
552                }
553                System.arraycopy( digest, 0, result, 0, length );
554            }
555            catch( NoSuchAlgorithmException e )
556            {
557                throw new CryptographyException( e );
558            }
559            return result;
560        }

571     * @param encryptMetadata The encryption metadata
572     *

574     *

576     * @throws IOException If there is an IO error.

579    public final byte[] computeUserPassword(
580            byte[] password,
581            byte[] o,
582            int permissions,
583            byte[] id,
584            int encRevision,
585            int length,
586            boolean encryptMetadata)
587            throws CryptographyException, IOException
588        {
589            ByteArrayOutputStream result = new ByteArrayOutputStream();
590            //STEP 1
591            byte[] encryptionKey = computeEncryptedKey( password, o, permissions, id, encRevision, length, encryptMetadata );
592 
593            if( encRevision == 2 )
594            {
595                //STEP 2
596                rc4.setKey( encryptionKey );
597                rc4.write( ENCRYPT_PADDING, result );
598            }
599            else if( encRevision == 3 || encRevision == 4 )
600            {
601                try
602                {
603                    //STEP 2
604                    MessageDigest md = MessageDigest.getInstance("MD5");
605                    //md.update( truncateOrPad( password ) );
606                    md.update( ENCRYPT_PADDING );
607 
608                    //STEP 3
609                    md.update( id );
610                    result.write( md.digest() );
611 
612                    //STEP 4 and 5
613                    byte[] iterationKey = new byte[ encryptionKey.length ];
614                    for( int i=0; i<20; i++ )
615                    {
616                        System.arraycopy( encryptionKey, 0, iterationKey, 0, iterationKey.length );
617                        for( int j=0; j< iterationKey.length; j++ )
618                        {
619                            iterationKey[j] = (byte)(iterationKey[j] ^ i);
620                        }
621                        rc4.setKey( iterationKey );
622                        ByteArrayInputStream input = new ByteArrayInputStream( result.toByteArray() );
623                        result.reset();
624                        rc4.write( input, result );
625                    }
626 
627                    //step 6
628                    byte[] finalResult = new byte[32];
629                    System.arraycopy( result.toByteArray(), 0, finalResult, 0, 16 );
630                    System.arraycopy( ENCRYPT_PADDING, 0, finalResult, 16, 16 );
631                    result.reset();
632                    result.write( finalResult );
633                }
634                catch( NoSuchAlgorithmException e )
635                {
636                    throw new CryptographyException( e );
637                }
638            }
639            return result.toByteArray();
640        }

703    // steps (a) to (d) of "Algorithm 3: Computing the encryption dictionary's O (owner password) value".
704    private byte[] computeRC4key(byte[] ownerPassword, int encRevision, int length)
705            throws NoSuchAlgorithmException, CryptographyException
706    {
707        if (encRevision == 2 && length != 5)
708        {
709            throw new CryptographyException(
710                    "Error: Expected length=5 actual=" + length);
711        }
712 
713        //3.3 STEP a
714        byte[] ownerPadded = truncateOrPad(ownerPassword);
715 
716        //3.3 STEP b
717        MessageDigest md = MessageDigest.getInstance("MD5");
718        md.update(ownerPadded);
719        byte[] digest = md.digest();
720 
721        //3.3 STEP c
722        if (encRevision == 3 || encRevision == 4)
723        {
724            for (int i = 0; i < 50; i++)
725            {
726                // this deviates from the spec - however, omitting the length
727                // parameter prevents the file to be opened in Adobe Reader
728                // with the owner password when the key length is 40 bit (= 5 bytes)
729                md.reset();
730                md.update(digest, 0, length);
731                digest = md.digest();
732            }
733        }
734 
735        //3.3 STEP d
736        byte[] rc4Key = new byte[(int) length];
737        System.arraycopy(digest, 0, rc4Key, 0, (int) length);
738        return rc4Key;
739    }

743     *

745     *
746     * @return The padded or truncated password.
747     */
748    private final byte[] truncateOrPad( byte[] password )
749    {
750        byte[] padded = new byte[ ENCRYPT_PADDING.length ];
751        int bytesBeforePad = Math.min( password.length, padded.length );

753        System.arraycopy( ENCRYPT_PADDING, 0, padded, bytesBeforePad, ENCRYPT_PADDING.length-bytesBeforePad );
754        return padded;
