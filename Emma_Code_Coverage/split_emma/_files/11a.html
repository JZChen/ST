515    {
516        if (clippingWindingRule > -1)
517        {
518            PDGraphicsState graphicsState = getGraphicsState();
519            GeneralPath clippingPath = (GeneralPath)getLinePath().clone(); // TODO do we really need to clone this? isn't the line path reset anyway?
520            clippingPath.setWindingRule(clippingWindingRule);
521            // If there is already set a clipping path, we have to intersect the new with the existing one
522            if (graphicsState.getCurrentClippingPath() != null) 
523            {
524                Area currentArea = new Area(getGraphicsState().getCurrentClippingPath());
525                Area newArea = new Area(clippingPath);
526                currentArea.intersect(newArea);
527                graphicsState.setCurrentClippingPath(currentArea);
528            }
529            else 
530            {
531                graphicsState.setCurrentClippingPath(clippingPath);
532            }
533            clippingWindingRule = -1;
534        }
535        getLinePath().reset();
536    }

567     * Fill with Shading.  Called by SHFill operator.
568     *
569     * @param shadingName  The name of the Shading Dictionary to use for this fill instruction.
570     *
571     * @throws IOException If there is an IO error while shade-filling the clipping area.
572     */
573    public void shFill(COSName shadingName) throws IOException
574    {
575        PDShadingResources shading = getResources().getShadings().get(shadingName.getName());
576        LOG.debug("Shading = " + shading.toString());
577        int shadingType = shading.getShadingType();
578        Matrix ctm = getGraphicsState().getCurrentTransformationMatrix();
579        Paint paint = null;
580        switch (shadingType)
581        {
582            case 1:
583                paint = new Type1ShadingPaint((PDShadingType1)shading, ctm, pageSize.height);
584                break;
585            case 2:
586                paint = new AxialShadingPaint((PDShadingType2)shading, ctm, pageSize.height);
587                break;
588            case 3:
589                paint = new RadialShadingPaint((PDShadingType3)shading, ctm, pageSize.height);
590                break;
591            case 4:
592                paint = new Type4ShadingPaint((PDShadingType4)shading, ctm, pageSize.height);
593                break;
594            case 5:
595                paint = new Type5ShadingPaint((PDShadingType5)shading, ctm, pageSize.height);
596                break;
597            case 6:
598                paint = new Type6ShadingPaint((PDShadingType6)shading, ctm, pageSize.height);
599                break;
600            case 7:
601                paint = new Type7ShadingPaint((PDShadingType7)shading, ctm, pageSize.height);
602                break;
603            default:
604                throw new IOException("Invalid ShadingType " + shadingType + " for Shading " + shadingName);
605        }
606        graphics.setComposite(getGraphicsState().getNonStrokeJavaComposite());
607        graphics.setPaint(paint);
608        graphics.setRenderingHint( RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_OFF );
609        graphics.setClip(null); // PDFBOX-2153 don't use obsolete clipping path
610        graphics.fill( getGraphicsState().getCurrentClippingPath() );
611    }

704    private float transformWidth(float width)
705    {
706        Matrix ctm = getGraphicsState().getCurrentTransformationMatrix();
707 
708        if (ctm == null)
709        {
710            // TODO does the CTM really need to use null?
711            return width;
712        }
713 
714        float x = ctm.getValue(0, 0) + ctm.getValue(1, 0);
715        float y = ctm.getValue(0, 1) + ctm.getValue(1, 1);
716        return width * (float) Math.sqrt((x * x + y * y) * 0.5);
717    }
