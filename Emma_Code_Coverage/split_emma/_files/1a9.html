506    /**
507     * This will parse the next object from the stream and add it to
508     * the local state.
509     *
510     * @return Returns true if the processed object had an endOfFile marker
511     *
512     * @throws IOException If an IO error occurs.
513     */
514    private boolean parseObject() throws IOException
515    {
516        long currentObjByteOffset = pdfSource.getOffset();
517        boolean isEndOfFile = false;
518        skipSpaces();
519        //peek at the next character to determine the type of object we are parsing
520        char peekedChar = (char)pdfSource.peek();
521 
522        //ignore endobj and endstream sections.
523        while( peekedChar == 'e' )
524        {
525            //there are times when there are multiple endobj, so lets
526            //just read them and move on.
527            readString();
528            skipSpaces();
529            currentObjByteOffset = pdfSource.getOffset();
530            peekedChar = (char)pdfSource.peek();
531        }
532        if( pdfSource.isEOF())
533        {
534            //"Skipping because of EOF" );
535            //end of file we will return a false and call it a day.
536        }
537        //xref table. Note: The contents of the Xref table are currently ignored
538        else if( peekedChar == 'x')
539        {
540            parseXrefTable( currentObjByteOffset );
541        }
542        // Note: startxref can occur in either a trailer section or by itself
543        else if (peekedChar == 't' || peekedChar == 's')
544        {
545            if(peekedChar == 't')
546            {
547                parseTrailer();
548                peekedChar = (char)pdfSource.peek();
549            }
550            if (peekedChar == 's')
551            {
552                parseStartXref();
553                // readString() calls skipSpaces() will skip comments... that's
554                // bad for us b/c the %%EOF flag is a comment
555                while(isWhitespace(pdfSource.peek()) && !pdfSource.isEOF())
556                {
557                    pdfSource.read(); // read (get rid of) all the whitespace
558                }
559                String eof = "";
560                if(!pdfSource.isEOF())
561                {
562                    eof = readLine(); // if there's more data to read, get the EOF flag
563                }
564 
565                // verify that EOF exists (see PDFBOX-979 for documentation on special cases)
566                if(!"%%EOF".equals(eof)) 
567                {
568                    if(eof.startsWith("%%EOF")) 
569                    {
570                        // content after marker -> unread with first space byte for read newline
571                        pdfSource.unread(SPACE_BYTE); // we read a whole line; add space as newline replacement
572                        pdfSource.unread(eof.substring(5).getBytes("ISO-8859-1"));
573                    } 
574                    else 
575                    {
576                        // PDF does not conform to spec, we should warn someone
577                        LOG.warn("expected='%%EOF' actual='" + eof + "'");
578                        // if we're not at the end of a file, just put it back and move on
579                        if(!pdfSource.isEOF()) 
580                        {
581                            pdfSource.unread( SPACE_BYTE ); // we read a whole line; add space as newline replacement
582                            pdfSource.unread(eof.getBytes("ISO-8859-1"));
583                        }
584                    }
585                }
586                isEndOfFile = true;
587            }
588        }
589        //we are going to parse an normal object
590        else
591        {
592            long number = -1;
593            int genNum;
594            String objectKey;
595            boolean missingObjectNumber = false;
596            try
597            {
598                char peeked = (char)pdfSource.peek();
599                if( peeked == '<' )
600                {
601                    missingObjectNumber = true;
602                }
603                else
604                {
605                    number = readObjectNumber();
606                }
607            }
608            catch( IOException e )
609            {
610                //ok for some reason "GNU Ghostscript 5.10" puts two endobj
611                //statements after an object, of course this is nonsense
612                //but because we want to support as many PDFs as possible
613                //we will simply try again
614                number = readObjectNumber();
615            }
616            if( !missingObjectNumber )
617            {
618                skipSpaces();
619                genNum = readGenerationNumber();
620 
621                objectKey = readString( 3 );
622                //System.out.println( "parseObject() num=" + number +
623                //" genNumber=" + genNum + " key='" + objectKey + "'" );
624                if( !objectKey.equals( "obj" ) )
625                {
626                    if (!isContinueOnError(null) || !objectKey.equals("o")) 
627                    {
628                        throw new IOException("expected='obj' actual='" + objectKey + "' " + pdfSource);
629                    }
630                    //assume that "o" was meant to be "obj" (this is a workaround for
631                    // PDFBOX-773 attached PDF Andersens_Fairy_Tales.pdf).
632                }
633            }
634            else
635            {
636                number = -1;
637                genNum = -1;
638            }
639 
640            skipSpaces();
641            COSBase pb = parseDirObject();
642            String endObjectKey = readString();
643 
644            if( endObjectKey.equals( "stream" ) )
645            {
646                pdfSource.unread( endObjectKey.getBytes("ISO-8859-1") );
647                pdfSource.unread( ' ' );
648                if( pb instanceof COSDictionary )
649                {
650                    pb = parseCOSStream( (COSDictionary)pb, getDocument().getScratchFile() );
651 
652                    // test for XRef type
653                    final COSStream strmObj = (COSStream) pb;
654                    
655                    // remember streams without length to check them later
656                    COSBase streamLength = strmObj.getItem(COSName.LENGTH);
657                    int length = -1;
658                    if (streamLength instanceof COSNumber)
659                    {
660                        length = ((COSNumber) streamLength).intValue();
661                    }
662                    if (length == -1)
663                    {
664                        streamLengthCheckSet.add(strmObj);
665                    }
666                    
667                    final COSName objectType = (COSName)strmObj.getItem( COSName.TYPE );
668                    if( objectType != null && objectType.equals( COSName.XREF ) )
669                    {
670                        // XRef stream
671                        parseXrefStream( strmObj, currentObjByteOffset );
672                    }
673                }
674                else
675                {
676                    // this is not legal
677                    // the combination of a dict and the stream/endstream forms a complete stream object
678                    throw new IOException("stream not preceded by dictionary");
679                }
680                skipSpaces();
681                endObjectKey = readLine();
682            }
683 
684            COSObjectKey key = new COSObjectKey( number, genNum );
685            COSObject pdfObject = document.getObjectFromPool( key );
686            if(pdfObject.getObject() == null)
687            {
688                pdfObject.setObject(pb);
689            }
690            /*
691             * If the object we returned already has a baseobject, then we have a conflict
692             * which we will resolve using information after we parse the xref table.
693             */
694            else
695            {
696                addObjectToConflicts(currentObjByteOffset, key, pb);
697            }
698 
699            if( !endObjectKey.equals( "endobj" ) )
700            {
701                if (endObjectKey.startsWith( "endobj" ) )
702                {
703                    /*
704                     * Some PDF files don't contain a new line after endobj so we
705                     * need to make sure that the next object number is getting read separately
706                     * and not part of the endobj keyword. Ex. Some files would have "endobj28"
707                     * instead of "endobj"
708                     */
709                    pdfSource.unread( SPACE_BYTE ); // add a space first in place of the newline consumed by readline()
710                    pdfSource.unread( endObjectKey.substring( 6 ).getBytes("ISO-8859-1") );
711                }
712                else if(endObjectKey.trim().endsWith("endobj"))
713                {
714                    /*
715                     * Some PDF files contain junk (like ">> ", in the case of a PDF
716                     * I found which was created by Exstream Dialogue Version 5.0.039)
717                     * in which case we ignore the data before endobj and just move on
718                     */
719                    LOG.warn("expected='endobj' actual='" + endObjectKey + "' ");
720                }
721                else if( !pdfSource.isEOF() )
722                {
723                    //It is possible that the endobj is missing, there
724                    //are several PDFs out there that do that so. Unread
725                    //and assume that endobj was missing
726                    pdfSource.unread( SPACE_BYTE ); // add a space first in place of the newline consumed by readline()
727                    pdfSource.unread( endObjectKey.getBytes("ISO-8859-1") );
728                }
729            }
730            skipSpaces();
731        }
732        return isEndOfFile;
733    }

735   /**
736    * Adds a new ConflictObj to the conflictList.
737    * @param offset the offset of the ConflictObj
738    * @param key The COSObjectKey of this object
739    * @param pb The COSBase of this conflictObj

741    */

748        ConflictObj conflictObj = new ConflictObj(offset, key, obj);
749        conflictList.add(conflictObj);
750    }

752    /**
753     * This will parse the startxref section from the stream.
754     * The startxref value is ignored.
755     *
756     * @return false on parsing error
757     * @throws IOException If an IO error occurs.
758     */
759    protected boolean parseStartXref() throws IOException
760    {
761        if(pdfSource.peek() != 's')
762        {
763            return false;
764        }
765        String startXRef = readString();
766        if( !startXRef.trim().equals( "startxref" ) )
767        {
768            return false;
769        }
770        skipSpaces();
771        /* This integer is the byte offset of the first object referenced by the xref or xref stream
772         * Needed for the incremental update (PREV)
773         */
774        getDocument().setStartXref(readLong());
775        return true;
776    }

779    /**
780     * This will parse the xref table from the stream and add it to the state
781     * The XrefTable contents are ignored.
782     * @param startByteOffset the offset to start at
783     * @return false on parsing error
784     * @throws IOException If an IO error occurs.
785     */
786    protected boolean parseXrefTable( long startByteOffset ) throws IOException
787    {
788        if(pdfSource.peek() != 'x')
789        {
790            return false;
791        }
792        String xref = readString();
793        if( !xref.trim().equals( "xref" ) )
794        {
795            return false;
796        }
797        
798        // check for trailer after xref
799        String str = readString();
800        byte[] b = str.getBytes("ISO-8859-1");
801        pdfSource.unread(b, 0, b.length);
802        
803        // signal start of new XRef
804        xrefTrailerResolver.nextXrefObj( startByteOffset );
805 
806        if (str.startsWith("trailer"))
807        {
808            LOG.warn("skipping empty xref table");
809            return false;
810        }
811        
812        /*
813         * Xref tables can have multiple sections.
814         * Each starts with a starting object id and a count.
815         */
816        while(true)
817        {
818            long currObjID = readObjectNumber(); // first obj id
819            long count = readLong(); // the number of objects in the xref table
820            skipSpaces();
821            for(int i = 0; i < count; i++)
822            {
823                if(pdfSource.isEOF() || isEndOfName((char)pdfSource.peek()))
824                {
825                    break;
826                }
827                if(pdfSource.peek() == 't')
828                {
829                    break;
830                }
831                //Ignore table contents
832                String currentLine = readLine();
833                String[] splitString = currentLine.split("\\s");
834                if (splitString.length < 3)
835                {
836                    LOG.warn("invalid xref line: " + currentLine);
837                    break;
838                }
839                /* This supports the corrupt table as reported in
840                 * PDFBOX-474 (XXXX XXX XX n) */
841                if(splitString[splitString.length-1].equals("n"))
842                {
843                    try
844                    {
845                        long currOffset = Long.parseLong(splitString[0]);
846                        int currGenID = Integer.parseInt(splitString[1]);
847                        COSObjectKey objKey = new COSObjectKey(currObjID, currGenID);
848                        xrefTrailerResolver.setXRef(objKey, currOffset);
849                    }
850                    catch(NumberFormatException e)
851                    {
852                        throw new IOException(e.getMessage());
853                    }
854                }
855                else if(!splitString[2].equals("f"))
856                {
857                    throw new IOException("Corrupt XRefTable Entry - ObjID:" + currObjID);
858                }
859                currObjID++;
860                skipSpaces();
861            }
862            skipSpaces();
863            char c = (char)pdfSource.peek();
864            if(c < '0' || c > '9')
865            {
866                break;
867            }
868        }
869        return true;
870    }
871 
872    /**
873     * This will parse the trailer from the stream and add it to the state.
874     *
875     * @return false on parsing error
876     * @throws IOException If an IO error occurs.
877     */
878    protected boolean parseTrailer() throws IOException
879    {
880        if(pdfSource.peek() != 't')
881        {
882            return false;
883        }
884        //read "trailer"
885        String nextLine = readLine();
886        if( !nextLine.trim().equals( "trailer" ) )
887        {
888            // in some cases the EOL is missing and the trailer immediately
889            // continues with "<<" or with a blank character
890            // even if this does not comply with PDF reference we want to support as many PDFs as possible
891            // Acrobat reader can also deal with this.
892            if (nextLine.startsWith("trailer"))
893            {
894                byte[] b = nextLine.getBytes("ISO-8859-1");
895                int len = "trailer".length();
896                pdfSource.unread('\n');
897                pdfSource.unread(b, len, b.length-len);
898            }
899            else
900            {
901                return false;
902            }
903        }
904 
905        // in some cases the EOL is missing and the trailer continues with " <<"
906        // even if this does not comply with PDF reference we want to support as many PDFs as possible
907        // Acrobat reader can also deal with this.
908        skipSpaces();
909 
910        COSDictionary parsedTrailer = parseCOSDictionary();
911        xrefTrailerResolver.setTrailer( parsedTrailer );
912 
913        // The version can also be specified within the document /Catalog
914        readVersionInTrailer(parsedTrailer);
915 
916        skipSpaces();
917        return true;
918    }

920    /**
921     * The document catalog can also have a /Version parameter which overrides the version specified
922     * in the header if, and only if it is greater.
923     *
924     * @param parsedTrailer the parsed catalog in the trailer
925     */
926    protected void readVersionInTrailer(COSDictionary parsedTrailer)
927    {
928        COSObject root = (COSObject) parsedTrailer.getItem(COSName.ROOT);
929        if (root != null)
930        {
931            COSBase item = root.getItem(COSName.VERSION);
932            if (item instanceof COSName)
933            {
934                COSName version = (COSName) item;
935                float trailerVersion = Float.valueOf(version.getName());
936                if (trailerVersion > document.getVersion())
937                {
938                    document.setVersion(trailerVersion);
939                }
940            }
941            else if (item != null)
942            {
943                LOG.warn("Incorrect /Version entry is ignored: " + item);
944            }
945        }
946    }

951     * @param stream the stream to be read
952     * @param objByteOffset the offset to start at

960    /**
961     * Fills XRefTrailerResolver with data of given stream.
