495                    }

497                }
498            }
499            else
500            {

502                if (fd.getFontName() != null)

505                }
506                if (awtFont == null)
507                {
508                    log.info("Can't find the specified font " + fd.getFontName() );

510                    TrueTypeFont ttf = getExternalFontFile2( fd );

516                        }

518                        {

520                        }
521                    }
522                }
523            }

528                log.info("Using font "+awtFont.getName()+ " instead");
529                setIsFontSubstituted(true);
530            }
531        }

535    private byte[] rebuildTTF(PDFontDescriptorDictionary fd, InputStream inputStream) throws IOException
536    {
537        // this is one possible case of an incomplete subfont which leads to a font exception
538        if (getFontEncoding() instanceof WinAnsiEncoding)
539        {
540            TTFParser ttfParser = new TTFParser(true);
541            TrueTypeFont ttf = ttfParser.parseTTF(inputStream);
542            TTFSubFont ttfSub = new TTFSubFont(ttf, "PDFBox-Rebuild");
543            for (int i=getFirstChar();i<=getLastChar();i++)
544            {
545                ttfSub.addCharCode(i);
546            }
547            ByteArrayOutputStream baos = new ByteArrayOutputStream();
548            ttfSub.writeToStream(baos);
549            return baos.toByteArray();
550        }
551        return null;
552    }

565    /**
566     * Permit to load an external TTF Font program file
567     *
568     * Created by Pascal Allain
569     * Vertical7 Inc.
570     *
571     * @param fd The font descriptor currently used
572     * @return A PDStream with the Font File program, null if fd is null
573     * @throws IOException If the font is not found
574     */
575    private TrueTypeFont getExternalFontFile2(PDFontDescriptorDictionary fd)
576        throws IOException
577    {
578        TrueTypeFont retval = null;
579 
580        if ( fd != null )
581        {
582            String baseFont = getBaseFont();
583            // search for a suitable font in the local environment
584            retval = org.apache.fontbox.util.FontManager.findTTFont(baseFont);
585            // can't find any font, use fall back mapping
586            // NOTE: this fall back mechanism will be removed with PDFBox 2.0.0
587            if (retval == null)
588            {
589                String fontResource = externalFonts.getProperty( UNKNOWN_FONT );
590                if( (baseFont != null) &&
591                     (externalFonts.containsKey(baseFont)) )
592                {
593                    fontResource = externalFonts.getProperty(baseFont);
594                }
595                if( fontResource != null )
596                {
597                    retval = (TrueTypeFont)loadedExternalFonts.get( baseFont );
598                    if( retval == null )
599                    {
600                        TTFParser ttfParser = new TTFParser();
601                        InputStream fontStream = ResourceLoader.loadResource( fontResource );
602                        if( fontStream == null )
603                        {
604                            throw new IOException( "Error missing font resource '" + externalFonts.get(baseFont) + "'" );
605                        }
606                        retval = ttfParser.parseTTF( fontStream );
607                        loadedExternalFonts.put( baseFont, retval );
608                    }
609                }
610            }
611        }
612        return retval;
613    }

660    {
661        float width = super.getFontWidth(charCode);
662        if (width <= 0)
663        {
664            if (advanceWidths.containsKey(charCode))
665            {
666                width = advanceWidths.get(charCode);
667            }
668            else
669            {

671                try
672                {

674                    if (ttf != null)
675                    {

677                        width = ttf.getAdvanceWidth(code);
678                        int unitsPerEM = ttf.getUnitsPerEm();
679                        // do we have to scale the width
680                        if (unitsPerEM != 1000)

684                    }
685                }
686                catch (IOException exception)
687                {
688                    width = 250;
689                }

691            }
692        }

696    private int getGlyphcode(int code)
697    {
698        extractCMaps();
699        int result = 0;
700        if (getFontEncoding() != null && !isSymbolicFont())
701        {
702            try
703            {
704                String charactername = getFontEncoding().getName(code);
705                if (charactername != null)
706                {
707                    if (cmapWinUnicode != null)
708                    {
709                        String unicode = Encoding.getCharacterForName(charactername);
710                        if (unicode != null)
711                        {
712                            result = unicode.codePointAt(0);
713                        }
714                        result = cmapWinUnicode.getGlyphId(result);
715                    }
716                    else if (cmapMacintoshSymbol != null && MacOSRomanEncoding.INSTANCE.hasCodeForName(charactername))
717                    {
718                        result = MacOSRomanEncoding.INSTANCE.getCode(charactername);
719                        result = cmapMacintoshSymbol.getGlyphId(result);
720                    }
721                    else if (cmapWinSymbol != null)
722                    {
723                        // fallback scenario if the glyph can't be found yet
724                        // maybe the 3,0 cmap provides a suitable mapping
725                        // see PDFBOX-2091
726                        result = cmapWinSymbol.getGlyphId(code);
727                    }
728                }
729            }
730            catch (IOException exception)
731            {
732                log.error("Caught an exception getGlyhcode: " + exception);
733            }
734        }
735        else if (getFontEncoding() == null || isSymbolicFont())
736        {
737            if (cmapWinSymbol != null)
738            {
739                result = cmapWinSymbol.getGlyphId(code);
740                if (code >= 0 && code <= 0xFF)
741                {
742                    // the CMap may use one of the following code ranges,
743                    // so that we have to add the high byte to get the
744                    // mapped value
745                    if (result == 0)
746                    {
747                        // F000 - F0FF
748                        result = cmapWinSymbol.getGlyphId(code + START_RANGE_F000);
749                    }
750                    if (result == 0)
751                    {
752                        // F100 - F1FF
753                        result = cmapWinSymbol.getGlyphId(code + START_RANGE_F100);
754                    }
755                    if (result == 0)
756                    {
757                        // F200 - F2FF
758                        result = cmapWinSymbol.getGlyphId(code + START_RANGE_F200);
759                    }
760                }
761            }
762            else if (cmapMacintoshSymbol != null)
763            {
764                result = cmapMacintoshSymbol.getGlyphId(code);
765            }
766        }
767        return result;
768    }

771     * extract all useful CMaps.
772     */
773    private void extractCMaps()
774    {
775        if (!cmapInitialized)
776        {
777            try 
778            {
779                getTTFFont();
780            }
781            catch(IOException exception)
782            {
783                log.error("Can't read the true type font", exception);
784            }
785            CMAPTable cmapTable = trueTypeFont.getCMAP();
786            if (cmapTable != null)
787            {
788                // get all relevant CMaps
789                CMAPEncodingEntry[] cmaps = cmapTable.getCmaps();
790                for (int i = 0; i < cmaps.length; i++)
791                {
792                    if (CMAPTable.PLATFORM_WINDOWS == cmaps[i].getPlatformId())
793                    {
794                        if (CMAPTable.ENCODING_UNICODE == cmaps[i].getPlatformEncodingId())
795                        {
796                            cmapWinUnicode = cmaps[i];
797                        }
798                        else if (CMAPTable.ENCODING_SYMBOL == cmaps[i].getPlatformEncodingId())
799                        {
800                            cmapWinSymbol = cmaps[i];
801                        }
802                    }
803                    else if (CMAPTable.PLATFORM_MACINTOSH == cmaps[i].getPlatformId())
804                    {
805                        if (CMAPTable.ENCODING_SYMBOL == cmaps[i].getPlatformEncodingId())
806                        {
807                            cmapMacintoshSymbol = cmaps[i];
808                        }
809                    }
810                }
811            }
812            cmapInitialized = true;
813        }
