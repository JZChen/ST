757    /**
758     * Escape special characters.
759     * 
760     * @param input the string to be escaped
761     * 
762     * @return the resulting string
763     */
764    private String escapeXML(String input)
765    {
766        StringBuilder escapedXML = new StringBuilder();
767        for (int i = 0; i < input.length(); i++)
768        {
769            char c = input.charAt(i);
770            switch (c)
771            {
772            case '<':
773                escapedXML.append("&lt;");
774                break;
775            case '>':
776                escapedXML.append("&gt;");
777                break;
778            case '\"':
779                escapedXML.append("&quot;");
780                break;
781            case '&':
782                escapedXML.append("&amp;");
783                break;
784            case '\'':
785                escapedXML.append("&apos;");
786                break;
787            default:
788                if (c > 0x7e)
789                {
790                    escapedXML.append("&#" + ((int) c) + ";");
791                }
792                else
793                {
794                    escapedXML.append(c);
795                }
796            }
797        }
798        return escapedXML.toString();
799    }
