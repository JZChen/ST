1009            throws IOException
1010    {
1011        float lineWidth = getLineWidth(tokens);
1012        float pos = 0.0f;
1013        if (parent.isMultiline())

1020            if (pdFont instanceof PDSimpleFont)

1022                // BJL 9/25/2004
1023                // This algorithm is a little bit of black magic. It does
