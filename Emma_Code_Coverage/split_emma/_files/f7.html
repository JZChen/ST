503     * @param dictionary
504     *            The dictionary to decrypt.
505     * @param objNum
506     *            The object number.
507     * @param genNum
508     *            The object generation number.
509     *
510     * @throws CryptographyException
511     *             If there is an error decrypting the document.
512     * @throws IOException
513     *             If there is an error creating a new string.
514     */
515    private void decryptDictionary(COSDictionary dictionary, long objNum, long genNum) throws CryptographyException,
516            IOException
517    {
518        for (Map.Entry<COSName, COSBase> entry : dictionary.entrySet())
519        {
520            COSBase value = entry.getValue();
521            // within a dictionary only the following kind of COS objects have to be decrypted
522            if (value instanceof COSString || value instanceof COSStream || value instanceof COSArray
523                    || value instanceof COSDictionary)
524            {
525                // if we are a signature dictionary and contain a Contents entry then
526                // we don't decrypt it.
527                if (!(entry.getKey().equals(COSName.CONTENTS) && value instanceof COSString && potentialSignatures
528                        .contains(dictionary)))
529                {
530                    decrypt(value, objNum, genNum);
531                }
532            }
533        }
534    }
