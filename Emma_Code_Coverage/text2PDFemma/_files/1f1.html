610    /**
611     * Determine if this TextPosition logically contains
612     * another (i.e. they overlap and should be rendered on top
613     * of each other).
614     * @param tp2 The other TestPosition to compare against
615     *
616     * @return True if tp2 is contained in the bounding box of this text.
617     */
618    public boolean contains( TextPosition tp2)
619    {
620        double thisXstart = getXDirAdj();
621        double thisXend = getXDirAdj() + getWidthDirAdj();
622 
623        double tp2Xstart = tp2.getXDirAdj();
624        double tp2Xend = tp2.getXDirAdj() + tp2.getWidthDirAdj();
625 
626        /*
627         * No X overlap at all so return as soon as possible.
628         */
629        if(tp2Xend <= thisXstart || tp2Xstart >= thisXend)
630        {
631            return false;
632        }
633        /*
634         * No Y overlap at all so return as soon as possible.
635         * Note: 0.0 is in the upper left and y-coordinate is
636         * top of TextPosition
637         */
638        if((tp2.getYDirAdj() + tp2.getHeightDir() < getYDirAdj()) ||
639                (tp2.getYDirAdj() > getYDirAdj() + getHeightDir()))
640        {
641            return false;
642        }
643        /* We're going to calculate the percentage of overlap. If its less
644         * than a 15% x-coordinate overlap then we'll return false because its negligible.
645         * .15 was determined by trial and error in the regression test files.
646         */
647        else if((tp2Xstart > thisXstart) && (tp2Xend > thisXend))
648        {
649            double overlap = thisXend - tp2Xstart;
650            double overlapPercent = overlap/getWidthDirAdj();
651            return (overlapPercent > .15);
652        }
653        else if((tp2Xstart < thisXstart) && (tp2Xend < thisXend))
654        {
655            double overlap = tp2Xend - thisXstart;
656            double overlapPercent = overlap/getWidthDirAdj();
657            return (overlapPercent > .15);
658        }
659        return true;
660    }

662    /**
663     * Merge a single character TextPosition into the current object.
664     * This is to be used only for cases where we have a diacritic that
665     * overlaps an existing TextPosition.  In a graphical display, we could
666     * overlay them, but for text extraction we need to merge them. Use the
667     * contains() method to test if two objects overlap.
668     *
669     * @param diacritic TextPosition to merge into the current TextPosition.
670     * @param normalize Instance of TextNormalize class to be used to normalize diacritic
671     */
672    public void mergeDiacritic(TextPosition diacritic, TextNormalize normalize)
673    {
674        if (diacritic.getCharacter().length() > 1)
675        {
676            return;
677        }
678 
679        float diacXStart = diacritic.getXDirAdj();
680        float diacXEnd = diacXStart + diacritic.widths[0];
681 
682        float currCharXStart = getXDirAdj();
683 
684        int strLen = str.length();
685        boolean wasAdded = false;
686 
687        for (int i = 0; i < strLen && !wasAdded; i++)
688        {
689            float currCharXEnd = currCharXStart + widths[i];
690 
691            /*
692             * This is the case where there is an overlap of the diacritic character with
693             * the current character and the previous character. If no previous character,
694             * just append the diacritic after the current one.
695             */
696            if(diacXStart < currCharXStart && diacXEnd <= currCharXEnd)
697            {
698                if(i == 0)
699                {
700                    insertDiacritic(i, diacritic, normalize);
701                }
702                else
703                {
704                    float distanceOverlapping1 = diacXEnd - currCharXStart;
705                    float percentage1 = distanceOverlapping1/widths[i];
706 
707                    float distanceOverlapping2 = currCharXStart - diacXStart;
708                    float percentage2 = distanceOverlapping2/widths[i-1];
709 
710                    if(percentage1 >= percentage2)
711                    {
712                        insertDiacritic(i, diacritic, normalize);
713                    }
714                    else
715                    {
716                        insertDiacritic(i-1, diacritic, normalize);
717                    }
718                }
719                wasAdded = true;
720            }
721            //diacritic completely covers this character and therefore we assume that
722            //this is the character the diacritic belongs to
723            else if(diacXStart < currCharXStart && diacXEnd > currCharXEnd)
724            {
725                insertDiacritic(i, diacritic, normalize);
726                wasAdded = true;
727            }
728            //Otherwise, The diacritic modifies this character because its completely
729            //contained by the character width
730            else if(diacXStart >= currCharXStart && diacXEnd <= currCharXEnd)
731            {
732                insertDiacritic(i, diacritic, normalize);
733                wasAdded = true;
734            }
735            /*
736             * Last character in the TextPosition so we add diacritic to the end
737             */
738            else if(diacXStart >= currCharXStart && diacXEnd > currCharXEnd && i == (strLen - 1))
739            {
740                insertDiacritic(i, diacritic, normalize);
741                wasAdded = true;
742            }
743            /*
744             * Couldn't find anything useful so we go to the next character in the
745             * TextPosition
746             */
747            currCharXStart += widths[i];
748        }
749    }
750    /**
751     * Inserts the diacritic TextPosition to the str of this TextPosition
752     * and updates the widths array to include the extra character width.
753     * @param i current character
754     * @param diacritic The diacritic TextPosition
755     * @param normalize Instance of TextNormalize class to be used to normalize diacritic
756     */
757    private void insertDiacritic(int i, TextPosition diacritic, TextNormalize normalize)
758    {
759        /* we add the diacritic to the right or left of the character
760         * depending on the direction of the character.  Note that this
761         * is only required because the text is currently stored in
762         * presentation order and not in logical order.
763         */
764        int dir = Character.getDirectionality(str.charAt(i));
765        StringBuffer buf = new StringBuffer();
766 
767        buf.append(str.substring(0,i));
768 
769        float[] widths2 = new float[widths.length+1];
770        System.arraycopy(widths, 0, widths2, 0, i);
771 
772        if ((dir == Character.DIRECTIONALITY_RIGHT_TO_LEFT)
773                || (dir == Character.DIRECTIONALITY_RIGHT_TO_LEFT_ARABIC)
774                || (dir == Character.DIRECTIONALITY_RIGHT_TO_LEFT_EMBEDDING)
775                || (dir == Character.DIRECTIONALITY_RIGHT_TO_LEFT_OVERRIDE))
776        {
777            buf.append(normalize.normalizeDiac(diacritic.getCharacter()));
778            widths2[i] = 0;
779            buf.append(str.charAt(i));
780            widths2[i+1] = widths[i];
781        }
782        else
783        {
784            buf.append(str.charAt(i));
785            widths2[i] = widths[i];
786            buf.append(normalize.normalizeDiac(diacritic.getCharacter()));
787            widths2[i+1] = 0;
788        }
789 
790        // Get the rest of the string
791        buf.append(str.substring(i+1, str.length()));
792        System.arraycopy(widths, i+1, widths2, i+2, widths.length-i-1);
793 
794        str = buf.toString();
795        widths = widths2;
796    }
